model User {
  id         String   @id @default(uuid())
  businessId String
  name       String
  lastName   String
  email      String
  password   String
  role       Role     @default(RECEPTION)
  active     Boolean  @default(true)
  createdAt  DateTime @default(now())
  updatedAt  DateTime @updatedAt

  business Business  @relation(fields: [businessId], references: [id])
  employee Employee?

  @@unique([businessId, email])
}

enum Role {
  ADMIN
  RECEPTION
  EMPLOYEE
}

model Sale {
  id         String @id @default(uuid())
  businessId String
  folio      Int    @default(autoincrement()) // Ticket #1, #2...

  // Relaciones
  clientId      String? // Opcional para "P√∫blico General"
  employeeId    String // Qui√©n atendi√≥/vendi√≥
  appointmentId String? @unique // Si viene de una cita, se liga aqu√≠

  // Totales
  subtotal Float
  tax      Float @default(0)
  discount Float @default(0)
  total    Float // Monto final a pagar

  status    SaleStatus @default(COMPLETED)
  createdAt DateTime   @default(now()) // HORA REAL DE LA VENTA
  updatedAt DateTime   @updatedAt

  business    Business     @relation(fields: [businessId], references: [id])
  client      Client?      @relation(fields: [clientId], references: [id])
  employee    Employee     @relation(fields: [employeeId], references: [id])
  appointment Appointment? @relation(fields: [appointmentId], references: [id])

  items    SaleItem[]
  payments Payment[]
}

model SaleItem {
  id          String  @id @default(uuid())
  saleId      String
  serviceId   String?
  description String // Nombre del servicio al momento de la venta
  price       Float // Precio hist√≥rico (por si el servicio sube luego)
  quantity    Int     @default(1)

  sale    Sale     @relation(fields: [saleId], references: [id])
  service Service? @relation(fields: [serviceId], references: [id])
}

enum SaleStatus {
  COMPLETED
  CANCELLED
  REFUNDED
}

model Service {
  id                String  @id @default(uuid())
  businessId        String
  categoryId        String
  name              String // Manicura, Pedicura, Ondas
  description       String?

  price             Float

  active    Boolean  @default(true)
  createdAt DateTime @default(now())

  business Business        @relation(fields: [businessId], references: [id])
  category ServiceCategory @relation(fields: [categoryId], references: [id])

  appointmentServices AppointmentService[]
  variants            ServiceVariant[]
  saleItems           SaleItem[]
}

model ServiceVariant {

  price     Float

  service Service @relation(fields: [serviceId], references: [id])
}

model Appointment {
  id            String            @id @default(uuid())
  businessId    String
  clientId      String?
  employeeId    String
  title         String
  start         DateTime
  end           DateTime
  allDay        Boolean           @default(false)
  status        AppointmentStatus @default(PENDING)
  notes         String?
  guestName     String?
  guestPhone    String?
  createdAt     DateTime          @default(now())
  // FINANZAS (Resumen)
  totalAmount   Float             @default(0) // El costo total del servicio (ej. $1000)
  paymentStatus PaymentState      @default(UNPAID) // Estado general de la deuda
  business      Business          @relation(fields: [businessId], references: [id])
  client        Client?           @relation(fields: [clientId], references: [id])
  employee      Employee          @relation(fields: [employeeId], references: [id])

  services AppointmentService[]
  payments Payment[]
  sale     Sale?
}

enum PaymentState {
  UNPAID // No ha pagado nada
  PARTIALLY_PAID // Dio anticipo pero debe el resto
  PAID // Pagado al 100%
  REFUNDED // Se le devolvi√≥ el dinero
}

enum AppointmentStatus {
  PENDING
  CONFIRMED
  CANCELLED
  COMPLETED
}

  price                Float
  duration             Int
  appointmentService   AppointmentService @relation(fields: [appointmentServiceId], references: [id])
  extra                ServiceExtra       @relation(fields: [extraId], references: [id])
}

model Payment {
  id            String        @id @default(uuid())
  appointmentId String?
  saleId        String?
  amount        Float // Cu√°nto pag√≥ en ESTA transacci√≥n
  method        PaymentMethod // CASH, CARD, TRANSFER, OTHER
  status        PaymentStatus @default(COMPLETED) // COMPLETED, PENDING, REFUNDED
  externalId    String? // ID de Stripe/PayPal o n√∫mero de autorizaci√≥n de la terminal

  createdAt DateTime @default(now())

  // Relaci√≥n inversa
  appointment Appointment @relation(fields: [appointmentId], references: [id])
  business    Business?   @relation(fields: [businessId], references: [id])
  businessId  String?
  sale        Sale        @relation(fields: [saleId], references: [id])
}

model CashClose {
  id          String   @id @default(uuid())
  businessId  String
  userId      String // Qui√©n cerr√≥ caja
  openingDate DateTime
  closingDate DateTime @default(now())

  expectedCash Float // Lo que el sistema dice que hay
  actualCash   Float // Lo que el humano cont√≥
  difference   Float // Faltante o sobrante

  business Business @relation(fields: [businessId], references: [id])
}

// Tus Enums para controlar los tipos
enum PaymentMethod {
  CASH
  CARD
  TRANSFER
}

enum PaymentStatus {
  PENDING
  COMPLETED
  FAILED
  REFUNDED
}
